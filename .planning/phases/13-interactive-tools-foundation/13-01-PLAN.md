<action>
Create `functions/api/routes/scouting/toa-proxy.ts` — a Hono route group that proxies requests to The Orange Alliance API. The proxy:

1. Accepts GET requests at `/api/scouting/toa/:path{.+}` 
2. Forwards to `https://theorangealliance.org/api/{path}`
3. Injects the `X-TOA-Key` header from environment variable `TOA_API_KEY`
4. Injects `X-Application-Origin` header with value `ARES-23247-ARESWEB`
5. Returns the JSON response, stripping CORS headers
6. Returns 502 with structured error on upstream failure

Key TOA endpoints to support:
- `/team/{teamNumber}` — team info
- `/team/{teamNumber}/results/{seasonKey}` — season results
- `/event/{eventKey}` — event info
- `/event/{eventKey}/matches` — match list
- `/event/{eventKey}/rankings` — event rankings
- `/event/{eventKey}/teams` — teams at event

```typescript
import { Hono } from "hono";
import type { HonoEnv } from "../../bindings";

const toa = new Hono<HonoEnv>();

toa.get("/:path{.+}", async (c) => {
  const path = c.req.param("path");
  const toaKey = c.env.TOA_API_KEY;
  if (!toaKey) return c.json({ error: "TOA_API_KEY not configured" }, 500);

  const url = `https://theorangealliance.org/api/${path}`;
  const res = await fetch(url, {
    headers: {
      "X-TOA-Key": toaKey,
      "X-Application-Origin": "ARES-23247-ARESWEB",
      "Content-Type": "application/json",
    },
  });

  if (!res.ok) {
    return c.json({ error: `TOA upstream error: ${res.status}`, status: res.status }, 502);
  }

  const data = await res.json();
  return c.json(data);
});

export default toa;
```

<action>
Create `functions/api/routes/scouting/ftcevents-proxy.ts` — proxy for the FIRST FTC Events API v2.0.

The FTC Events API uses Basic Auth with username (from `FTC_EVENTS_USERNAME` env) and a token (from `FTC_EVENTS_API_KEY` env).

Base URL: `https://ftc-api.firstinspires.org/v2.0`

Key endpoints:
- `/{season}/events` — list events for a season
- `/{season}/matches/{eventCode}` — match results
- `/{season}/rankings/{eventCode}` — rankings
- `/{season}/teams?eventCode={code}` — teams at event
- `/{season}/scores/{eventCode}/{tournamentLevel}` — detailed scores

```typescript
import { Hono } from "hono";
import type { HonoEnv } from "../../bindings";

const ftcEvents = new Hono<HonoEnv>();

ftcEvents.get("/:path{.+}", async (c) => {
  const path = c.req.param("path");
  const username = c.env.FTC_EVENTS_USERNAME;
  const apiKey = c.env.FTC_EVENTS_API_KEY;
  if (!username || !apiKey) return c.json({ error: "FTC Events API credentials not configured" }, 500);

  const authHeader = "Basic " + btoa(`${username}:${apiKey}`);
  const url = `https://ftc-api.firstinspires.org/v2.0/${path}`;
  
  const res = await fetch(url, {
    headers: {
      Authorization: authHeader,
      "Content-Type": "application/json",
    },
  });

  if (!res.ok) {
    return c.json({ error: `FTC Events upstream error: ${res.status}`, status: res.status }, 502);
  }

  const data = await res.json();
  return c.json(data);
});

export default ftcEvents;
```

```typescript
const BASE = "/api/scouting";

export interface TOATeam {
  team_key: string;
  team_number: number;
  team_name_short: string;
  team_name_long: string;
  robot_name: string;
  city: string;
  state_prov: string;
  country: string;
  rookie_year: number;
}

export interface TOARanking {
  rank: number;
  team_key: string;
  opr: number;
  np_opr: number;
  wins: number;
  losses: number;
  ties: number;
  highest_qual_score: number;
  ranking_points: number;
  qualifying_points: number;
  tie_breaker_points: number;
  played: number;
}

export interface TOAMatch {
  match_key: string;
  match_name: string;
  red_score: number;
  blue_score: number;
  participants: { team_key: string; station: number }[];
}

export interface AnalysisRequest {
  mode: "team_analysis" | "match_prediction" | "event_overview";
  teamNumber?: number;
  eventKey?: string;
  seasonKey: string;
  context: Record<string, unknown>;
}

export interface AnalysisResponse {
  markdown: string;
  model: string;
  tokensUsed?: number;
}

async function fetchJSON<T>(url: string): Promise<T> {
  const res = await fetch(url);
  if (!res.ok) throw new Error(`Scouting API error: ${res.status}`);
  return res.json();
}

export const scoutingApi = {
  // TOA endpoints
  getTeam: (teamNumber: number) => fetchJSON<TOATeam>(`${BASE}/toa/team/${teamNumber}`),
  getTeamResults: (teamNumber: number, seasonKey: string) => fetchJSON<unknown>(`${BASE}/toa/team/${teamNumber}/results/${seasonKey}`),
  getEventRankings: (eventKey: string) => fetchJSON<TOARanking[]>(`${BASE}/toa/event/${eventKey}/rankings`),
  getEventMatches: (eventKey: string) => fetchJSON<TOAMatch[]>(`${BASE}/toa/event/${eventKey}/matches`),
  getEventTeams: (eventKey: string) => fetchJSON<TOATeam[]>(`${BASE}/toa/event/${eventKey}/teams`),

  // FTC Events endpoints
  getSeasonEvents: (season: number) => fetchJSON<unknown>(`${BASE}/ftcevents/${season}/events`),
  getEventScores: (season: number, eventCode: string, level: string = "qual") =>
    fetchJSON<unknown>(`${BASE}/ftcevents/${season}/scores/${eventCode}/${level}`),

  // AI Analysis
  analyze: async (req: AnalysisRequest): Promise<AnalysisResponse> => {
    const res = await fetch(`${BASE}/analyze`, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify(req),
    });
    if (!res.ok) throw new Error(`Analysis API error: ${res.status}`);
    return res.json();
  },
};
```
</action>
