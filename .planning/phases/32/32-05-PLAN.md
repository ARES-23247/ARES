## Fix Pattern for Standard Hono Routes

Before:
```typescript
/* eslint-disable @typescript-eslint/ban-ts-comment */
// @ts-nocheck
import { Hono } from "hono";
import { AppEnv, getSessionUser } from "../middleware";

const authRouter = new Hono<AppEnv>();

authRouter.get("/auth-check", async (c) => { // Implicit 'any'
  const user = await getSessionUser(c);
  return c.json({ authenticated: !!user });
});
```

After:
```typescript
import { Hono } from "hono";
import type { Context } from "hono";
import { AppEnv, getSessionUser } from "../middleware";

const authRouter = new Hono<AppEnv>();

authRouter.get("/auth-check", async (c: Context<AppEnv>) => { // TYPED
  const user = await getSessionUser(c);
  return c.json({ authenticated: !!user });
});
```
